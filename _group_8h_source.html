152 // keep first 32 Byte of first signature of each transaction from last MAGIC_NUMBER_SIGNATURE_CACHE_MINUTES minutes
153 struct HalfSignature
154 {
155 HalfSignature(const char* signature) {
156 memcpy(&sign, signature, 32);
157 }
158 HalfSignature(const model::gradido::GradidoTransaction* transaction) {
159 auto sigPairs = transaction->getSignaturesfromSignatureMap(true);
160 if (sigPairs.size() == 0) {
161 throw std::runtime_error("[Group::addSignatureToCache] empty signatures");
162 }
163
164 memcpy(&sign, sigPairs[0]->data(), 32);
165 auto mm = MemoryManager::getInstance();
166 mm->releaseMemory(sigPairs[0]);
167 }
168 bool operator<(const HalfSignature& ob) const {
169 return
170 sign[0] < ob.sign[0] ||
171 (sign[0] == ob.sign[0] && sign[1] < ob.sign[1]) || (
172 sign[0] == ob.sign[0] &&
173 sign[1] == ob.sign[1] &&
174 sign[2] < ob.sign[2]
175 ) || (
176 sign[0] == ob.sign[0] &&
177 sign[1] == ob.sign[1] &&
178 sign[2] == ob.sign[2] &&
179 sign[3] < ob.sign[3]
180 );
181 }
182 //bool comp(a, b)
183 int64_t sign[4];
184 };
